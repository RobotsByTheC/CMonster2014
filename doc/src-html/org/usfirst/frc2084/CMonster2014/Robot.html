020/**
021 * The VM is configured to automatically run this class, and to call the
022 * functions corresponding to each mode, as described in the IterativeRobot
023 * documentation. If you change the name of this class or the package after
024 * creating this project, you must also update the manifest file in the resource
025 * directory.
026 */
027public class Robot extends IterativeRobot {
028
029    private Command autonomousCommand;
030    private Command funCommand;
031    private final SendableChooser autonomousChooser = new SendableChooser();
032    public static OI oi;
033    // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DECLARATIONS
034    public static DriveSubsystem driveSubsystem;
035    public static CompressorSubsystem compressorSubsystem;
036    public static SweeperSubsystem sweeperSubsystem;
037    public static CatcherSubsytem catcherSubsytem;
038    public static LedSubsystem ledSubsystem;
039    // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DECLARATIONS
040    /**
041     * This function is run when the robot is first started up and should be
042     * used for any initialization code.
043     */
044    public void robotInit() {
045        RobotMap.init();
046        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=CONSTRUCTORS
047        driveSubsystem = new DriveSubsystem();
048        compressorSubsystem = new CompressorSubsystem();
049        sweeperSubsystem = new SweeperSubsystem();
050        catcherSubsytem = new CatcherSubsytem();
051        ledSubsystem = new LedSubsystem();
052        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=CONSTRUCTORS
053        oi = new OI();
054        // instantiate the command used for the autonomous period
055        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=AUTONOMOUS
056        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=AUTONOMOUS
057        funCommand = new FunCommand();
058        funCommand.start();
059        autonomousChooser.addDefault("Front Left Goal", new FrontAutonomousCommandGroup(true));
060        autonomousChooser.addObject("Front Right Goal", new FrontAutonomousCommandGroup(false));
061        SmartDashboard.putData("Autonomous Mode", autonomousChooser);
062
063        // Make sure the DS laptop is not reporting vision to the robot
064        TargetTrackingCommunication.setAutonomousVisionRunning(false);
065
066        // Enable the camera on the DS laptop when the robot starts
067        TargetTrackingCommunication.setCameraEnabled(true);
068
069        Robot.ledSubsystem.sendCode(LedSubsystem.DISABLE_CODE);
070    }
071
072    public void autonomousInit() {
073//        // Last ditch effort to bring the camera up on the DS laptop, probably
074//        // is too late.
075//        TargetTrackingCommunication.setCameraEnabled(true);
076//        // Tell the DS laptop to starting detecting the hot target
077//        TargetTrackingCommunication.setAutonomousVisionRunning(true);
078//        Removed this and put it in the autonomous command
079//        Robot.driveSubsystem.getRobotDrive().resetGyro();
080
081        // Pick which autonomous mode to use.
082        Object selection = autonomousChooser.getSelected();
083        if (selection != null && selection instanceof Command) {
084            autonomousCommand = (Command) selection;
085            autonomousCommand.start();
086        } else {
087            System.out.println("No autonomous mode selected.");
088        }
089
090        Alliance alliance = DriverStation.getInstance().getAlliance();
091        if (alliance.value == Alliance.kBlue_val) {
092            Robot.ledSubsystem.sendCode(LedSubsystem.BLINK_BLUE_CODE);
093        } else if (alliance.value == Alliance.kRed_val) {
094            Robot.ledSubsystem.sendCode(LedSubsystem.BLINK_RED_CODE);
095        } else {
096            Robot.ledSubsystem.sendCode(LedSubsystem.BLINK_GREEN_CODE);
097        }
098    }
099
100    /**
101     * This function is called periodically during autonomous
102     */
103    public void autonomousPeriodic() {
104        Scheduler.getInstance().run();
105    }
106
107    public void teleopInit() {
108        Alliance alliance = DriverStation.getInstance().getAlliance();
109        if (alliance.value == Alliance.kBlue_val) {
110            Robot.ledSubsystem.sendCode(LedSubsystem.SOLID_BLUE_CODE);
111        } else if (alliance.value == Alliance.kRed_val) {
112            Robot.ledSubsystem.sendCode(LedSubsystem.SOLID_RED_CODE);
113        } else {
114            Robot.ledSubsystem.sendCode(LedSubsystem.SOLID_GREEN_CODE);
115        }
116
117        TargetTrackingCommunication.setAutonomousVisionRunning(false);
118        TargetTrackingCommunication.setCameraEnabled(false);
119        // This makes sure that the autonomous stops running when
120        // teleop starts running. If you want the autonomous to 
121        // continue until interrupted by another command, remove
122        // this line or comment it out.
123        if (autonomousCommand != null) {
124            autonomousCommand.cancel();
125        }
126        Robot.driveSubsystem.getMecanumDriveAlgorithm().resetGyro();
127    }
128
129    /**
130     * This function is called periodically during operator control
131     */
132    public void teleopPeriodic() {
133        Scheduler.getInstance().run();
134    }
135
136    /**
137     * This function called periodically during test mode
138     */
139    public void testPeriodic() {
140        LiveWindow.run();
141    }
142
143    public void disabledInit() {
144        Robot.ledSubsystem.sendCode(LedSubsystem.PULSE_GREEN_CODE);
145    }
146
147    public void disabledPeriodic() {
148        Scheduler.getInstance().run();
149    }
150}
