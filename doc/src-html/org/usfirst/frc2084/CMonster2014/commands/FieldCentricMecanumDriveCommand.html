007package org.usfirst.frc2084.CMonster2014.commands;
008
009import edu.wpi.first.wpilibj.Joystick;
010import edu.wpi.first.wpilibj.command.Command;
011import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
012import org.usfirst.frc2084.CMonster2014.Robot;
013import org.usfirst.frc2084.CMonster2014.RobotMap;
014
015/**
016 * The command that implements the control, filtering and scaling of our amazing
017 * Mecanum drive system.
018 *
019 * @author Ben Wolsieffer
020 */
021public class FieldCentricMecanumDriveCommand extends Command {
022
023    /**
024     * The joystick z-axis value below which the robot will not rotate. this is
025     * to prevent accidental small twists of the joystick from affecting its
026     * trajectory.
027     */
028    public double ROTATION_DEADBAND = 0.2;
029    /**
030     * The maximum speed that the robot is allowed to rotate at. The joystick
031     * value is scaled down to this value.
032     */
033    public static final double MAX_ROTATION = 0.5;
034
035    public FieldCentricMecanumDriveCommand() {
036        // This command drives, so it requires the drive subsytem.
037        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
038        requires(Robot.driveSubsystem);
039        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
040    }
041
042    /**
043     * Does nothing.
044     */
045    protected void initialize() {
046    }
047
048    /**
049     * Updates the robot's speed based on the joystick values.
050     */
051    protected void execute() {
052        // This is the joystick that we use as input for driving
053        Joystick driveJoystick = Robot.oi.getDriveJoystick();
054        // Invert the z-axis (rotation) and store it.
055        double z = -driveJoystick.getZ();
056        // This will hold the scaled rotation value. We scale down this value
057        // because otherwise the robot is too hard ot control with the joystick 
058        // twist and we don't need our full possible rotation speed (its pretty
059        // fast).
060        double scaledZ = z;
061        // we implemented a deadband in order to filter out small accidental twists
062        // of the stick. If the rotation value (z) is less than the deadband, 
063        // we don't rotate.
064        if (Math.abs(z) < ROTATION_DEADBAND) {
065            scaledZ = 0;
066        } else {
067            // Scale z down so it is never greater than MAX_ROTATION.
068            scaledZ += z < 0 ? ROTATION_DEADBAND : -ROTATION_DEADBAND;
069            scaledZ = (scaledZ / (1.0 - ROTATION_DEADBAND)) * MAX_ROTATION;
070        }
071        // Print out the rotation speed for debugging.
072        SmartDashboard.putNumber("Rotation Speed", scaledZ);
073        // Actually drive the robot using the joystick values for x and y and
074        // the scaled z value. The inversions are necessary because of the way
075        // the rest of the code is set. We shouldn't touch them until we have
076        // time to go through and make sure we can fix all the unnecessary
077        // inversions.
078        Robot.driveSubsystem.getMecanumDriveAlgorithm().mecanumDrive_Cartesian(
079                -driveJoystick.getX(),
080                -driveJoystick.getY(),
081                scaledZ,
082                RobotMap.driveSubsystemSteeringGyro.getAngle()
083        );
084    }
085
086    /**
087     * This command never ends on its own but it could be interrupted, for
088     * example if we reverted back to our failsafe driving mode.
089     *
090     * @return false
091     */
092    protected boolean isFinished() {
093        return false;
094    }
095
096    /**
097     * Stops the drive motors.
098     */
099    protected void end() {
100        Robot.driveSubsystem.getMecanumDriveAlgorithm().stop();
101    }
102
103    /**
104     * Stops the drive motors.
105     */
106    protected void interrupted() {
107        end();
108    }
109}
