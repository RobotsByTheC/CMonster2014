007package org.usfirst.frc2084.CMonster2014.commands;
008
009import edu.wpi.first.wpilibj.command.CommandGroup;
010
011/**
012 * CommandGroup that runs the driving command in parallel with the vision
013 * command. This needs to be separated into its own command group to allow it to
014 * run sequentially after the initial wait at the beginning of
015 * {@link FrontAutonomousCommandGroup} and before the sweeper expels the ball.
016 *
017 * @see FrontAutonomousCommandGroup
018 */
019public class FrontAutonomousDriveAndVisionCommandGroup extends CommandGroup {
020
021    /**
022     * Instantiates this command group to drive toward the specified wall. The
023     * parameter is simply passed through to the driving command.
024     *
025     * @param leftGoal left goal = true, right goal = false
026     */
027    public FrontAutonomousDriveAndVisionCommandGroup(boolean leftGoal) {
028        // Run these in parallel so that the DS laptop start processing vision 
029        // while the robot drives. This works because the vision targets are 
030        // still visible until the robot is very close to the wall.
031        addParallel(new FrontAutonomousDriveCommand(leftGoal));
032        addParallel(new WaitForVisionCommand());
033    }
034}
