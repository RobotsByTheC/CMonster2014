007package org.usfirst.frc2084.CMonster2014.commands;
008
009import edu.wpi.first.wpilibj.ADXL345_I2C;
010import edu.wpi.first.wpilibj.DriverStationLCD;
011import edu.wpi.first.wpilibj.command.Command;
012import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
013import org.usfirst.frc2084.CMonster2014.RobotMap;
014import org.usfirst.frc2084.CMonster2014.TargetTrackingCommunication;
015
016/**
017 * Command that is always running, calculates pi and report debugging values.
018 *
019 * @author Ben Wolsieffer
020 */
021public class FunCommand extends Command {
022
023    /**
024     * Stores the current value of pi. Used by the pi calculating algorithm.
025     */
026    private double pi;
027    /**
028     * Stores the value that is 1/4 of pi. Used by the pi calculating algorithm.
029     * This is the actual value that the algorithm calculates.
030     */
031    private double pi4;
032    /**
033     * This value gets toggled each time the pi calculating algorithm runs. Used
034     * by
035     */
036    private boolean piPositive = true;
037    private long piIndex = 1;
038
039    public FunCommand() {
040        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
041        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
042        // Make sure this command still reports data and caclualtes pi when the 
043        // robot is disabled.
044        setRunWhenDisabled(true);
045    }
046
047    /**
048     * Does nothing.
049     */
050    protected void initialize() {
051    }
052
053    /**
054     * Calculates Pi :), and prints out debugging data to the SmartDashboard.
055     */
056    protected void execute() {
057        // Calculate Pi!!!! (http://en.wikipedia.org/wiki/Leibniz_formula_for_%CF%80)
058        // This method is really slow
059        pi4 += (piPositive ? 1.0 : -1.0) / piIndex;
060        piPositive = !piPositive;
061        piIndex += 2.0;
062        pi = pi4 * 4.0;
063        // Print out Pi to the DS LCD (not the SmartDashboard) because it can 
064        // display more digits
065        DriverStationLCD.getInstance().println(DriverStationLCD.Line.kUser1, 1, "Pi: " + pi);
066        DriverStationLCD.getInstance().updateLCD();
067        // Pi
068        // Report misc values
069        SmartDashboard.putNumber("Gyro Rate", RobotMap.driveSubsystemSteeringGyro.getRate());
070        SmartDashboard.putNumber("Gyro Angle", RobotMap.driveSubsystemSteeringGyro.getAngle());
071        ADXL345_I2C.AllAxes accel = RobotMap.driveSubsystemAccelerometer.getAccelerations();
072        SmartDashboard.putNumber("X Acceleration", accel.XAxis);
073        SmartDashboard.putNumber("Y Acceleration", accel.YAxis);
074        String targetState;
075        switch (TargetTrackingCommunication.getState().value) {
076            case TargetTrackingCommunication.State.HOT_VALUE:
077                targetState = "Hot";
078                break;
079            case TargetTrackingCommunication.State.NOT_HOT_VALUE:
080                targetState = "Not hot";
081                break;
082            default:
083            case TargetTrackingCommunication.State.UNKNOWN_VALUE:
084                targetState = "Unknown";
085                break;
086        }
087        SmartDashboard.putString("Goal State", targetState);
088
089        SmartDashboard.putNumber("Encoder Distance", RobotMap.driveSubsystemRearRightEncoder.getDistance());
090        SmartDashboard.putNumber("Temperature", RobotMap.driveSubsystemSteeringGyroTemp.getTemp());
091    }
092
093    // Make this return true when this Command no longer needs to run execute()
094    protected boolean isFinished() {
095        return false;
096    }
097
098    // Called once after isFinished returns true
099    protected void end() {
100    }
101
102    // Called when another command which requires one or more of the same
103    // subsystems is scheduled to run
104    protected void interrupted() {
105    }
106}
