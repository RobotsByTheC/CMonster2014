007package org.usfirst.frc2084.CMonster2014.commands;
008
009import org.usfirst.frc2084.CMonster2014.Robot;
010
011/**
012 * Command that extends the sweeper. It takes a certain amount of time for the
013 * pneumatics to move so this command extends {@link TimedCommand} which keeps
014 * it from ending immediately.
015 *
016 * @author Ben Wolsieffer
017 */
018public class SweeperExtendCommand extends TimedCommand {
019
020    public SweeperExtendCommand() {
021        // I estimated that it takes around 0.75 seconds to open the sweeper, so
022        // that's how long this command takes.
023        super(0.75);
024        // This command uses the sweeper assembly so it requires the sweeper 
025        // subsystem.
026        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
027        requires(Robot.sweeperSubsystem);
028        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=REQUIRES
029    }
030
031    /**
032     * Extends the sweeper.
033     */
034    protected void run() {
035        Robot.sweeperSubsystem.extend();
036    }
037
038    /**
039     * Does nothing.
040     */
041    protected void end() {
042    }
043
044    /**
045     * Does nothing.
046     */
047    protected void interrupted() {
048    }
049}
