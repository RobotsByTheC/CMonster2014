007package org.usfirst.frc2084.CMonster2014.drive;
008
009/**
010 * This class acts as a bridge between a HID (joystick, gamepad) and a
011 * {@link DriveController}. It takes input and uses it to control the robot
012 * through different algorithms. This class cannot be used directly and must be
013 * extended to provide functionality. A drive algorithm does not have to worry
014 * about where the wheels are in most cases, because this is taken care of by
015 * the {@link DriveController}.
016 *
017 * @see ArcadeDriveAlgorithm
018 * @see MecanumDriveAlgorithm
019 * @see DriveController
020 *
021 * @author Ben Wolsieffer
022 */
023public abstract class DriveAlgorithm {
024
025    protected DriveController controller;
026
027    /**
028     * Creates a new {@link DriveAlgorithm} that controls the specified
029     * {@link DriveController}.
030     *
031     * @param controller the drive controller to use
032     */
033    public DriveAlgorithm(DriveController controller) {
034        this.controller = controller;
035    }
036
037    /**
038     * Stops the robot. This just calls {@link DriveController#stopMotor()}.
039     */
040    public void stop() {
041        controller.stopMotor();
042    }
043}
