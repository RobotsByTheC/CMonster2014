007package org.usfirst.frc2084.CMonster2014.drive;
008
009/**
010 * Drive controller for robots with four wheels. Allows control over each side
011 * together or each wheel independently.
012 *
013 * @author Ben Wolsieffer
014 */
015public class FourWheelDriveController extends DriveController {
016
017    protected WheelController frontLeftWheel;
018    protected WheelController frontRightWheel;
019    protected WheelController rearLeftWheel;
020    protected WheelController rearRightWheel;
021
022    /**
023     * Creates a new {@link FourWheelDriveController} with the specified
024     * {@link WheelController}s.
025     *
026     * @param frontLeftWheel the front left {@link WheelController}
027     * @param frontRightWheel the front right {@link WheelController}
028     * @param rearLeftWheel the rear left {@link WheelController}
029     * @param rearRightWheel the rear right {@link WheelController}
030     */
031    public FourWheelDriveController(WheelController frontLeftWheel, WheelController frontRightWheel, WheelController rearLeftWheel, WheelController rearRightWheel) {
032        this.frontLeftWheel = frontLeftWheel;
033        this.frontRightWheel = frontRightWheel;
034        this.rearLeftWheel = rearLeftWheel;
035        this.rearRightWheel = rearRightWheel;
036    }
037
038    /**
039     * Set the left and right speeds of the robot using the specified values.
040     * For the {@link FourWheelDriveController}, {@code leftSpeed} sets both
041     * left {@link WheelController}s and {@code rightSpeed} sets both right
042     * {@link WheelController}s.
043     *
044     * @param leftSpeed {@inheritDoc}
045     * @param rightSpeed {@inheritDoc}
046     */
047    public void drive(double leftSpeed, double rightSpeed) {
048        frontLeftWheel.set(leftSpeed);
049        rearLeftWheel.set(leftSpeed);
050        frontRightWheel.set(rightSpeed);
051        rearRightWheel.set(rightSpeed);
052
053        // Make sure to feed the watchdog!
054        safetyHelper.feed();
055    }
056
057    /**
058     * Sets the speeds of all four {@link WheelController}s independently.
059     *
060     * @param frontLeftSpeed the speed of the front left wheel
061     * @param frontRightSpeed the speed of the front right wheel
062     * @param rearLeftSpeed the speed of the rear left wheel
063     * @param rearRightSpeed the speed of the rear right wheel
064     */
065    public void drive(double frontLeftSpeed, double frontRightSpeed, double rearLeftSpeed, double rearRightSpeed) {
066        frontLeftWheel.set(frontLeftSpeed);
067        frontRightWheel.set(frontRightSpeed);
068        rearLeftWheel.set(rearLeftSpeed);
069        rearRightWheel.set(rearRightSpeed);
070
071        // Make sure to feed the watchdog!
072        safetyHelper.feed();
073    }
074
075    /**
076     * Stops the robot.
077     */
078    public void stopMotor() {
079        frontLeftWheel.set(0);
080        rearLeftWheel.set(0);
081        frontRightWheel.set(0);
082        rearRightWheel.set(0);
083    }
084
085    public String getDescription() {
086        return "Four Wheel Drive Controller";
087    }
088}
