009import org.usfirst.frc2084.CMonster2014.RobotMap;
010import edu.wpi.first.wpilibj.*;
011import edu.wpi.first.wpilibj.command.Subsystem;

013/**
014 * The subsystem that communicates with the Arduino to control the LED strip. It
015 * uses six DIO pins to form a six bit number. Each number codes for a different
016 * led pattern.
017 */
018public class LedSubsystem extends Subsystem {
019
020    // LED strip pattern defintions (pretty self explanatory)
021    public static final int DISABLE_CODE = 0x3F;
022    public static final int SOLID_RED_CODE = 0x0;
023    public static final int SOLID_BLUE_CODE = 0x1;
024    public static final int SOLID_GREEN_CODE = 0x2;
025    public static final int BLINK_RED_CODE = 0x3;
026    public static final int BLINK_BLUE_CODE = 0x4;
027    public static final int BLINK_GREEN_CODE = 0x5;
028    public static final int PULSE_RED_CODE = 0x6;
029    public static final int PULSE_BLUE_CODE = 0x7;
030    public static final int PULSE_GREEN_CODE = 0x8;
031    public static final int CONVERGE_RED_CODE = 0x9;
032    public static final int CONVERGE_BLUE_CODE = 0xA;
033    public static final int CONVERGE_GREEN_CODE = 0xB;
034    public static final int DIVERGE_RED_CODE = 0xC;
035    public static final int DIVERGE_BLUE_CODE = 0xD;
036    public static final int DIVERGE_GREEN_CODE = 0xE;
037    public static final int RANDOM_PATTERN_CODE = 0xF;
038
039    // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DECLARATIONS
040    DigitalOutput pin0 = RobotMap.ledSubsystemPin0;
041    DigitalOutput pin1 = RobotMap.ledSubsystemPin1;
042    DigitalOutput pin2 = RobotMap.ledSubsystemPin2;
043    DigitalOutput pin3 = RobotMap.ledSubsystemPin3;
044    DigitalOutput pin4 = RobotMap.ledSubsystemPin4;
045    DigitalOutput pin5 = RobotMap.ledSubsystemPin5;
046    // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DECLARATIONS
047
048    public void initDefaultCommand() {
049        // BEGIN AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DEFAULT_COMMAND
050        // END AUTOGENERATED CODE, SOURCE=ROBOTBUILDER ID=DEFAULT_COMMAND
051    }
052
053    /**
054     * Sends the specified code to the DIO pins and to the Arduino.
055     *
056     * @param code the code to send
057     */
058    public void sendCode(int code) {
059        // The mask initially starts at one to get the value of the first bit 
060        // the first time the loop runs.
061        byte mask = 0x1;
062        // Loops through the first six bits in the code and sets the apporiate 
063        // DIO to its value.
064        for (int i = 0; i < 6; i++) {
065            DigitalOutput out;
066            switch (i) {
067                case 0:
068                    out = pin0;
069                    break;
070                case 1:
071                    out = pin1;
072                    break;
073                case 2:
074                    out = pin2;
075                    break;
076                case 3:
077                    out = pin3;
078                    break;
079                case 4:
080                    out = pin4;
081                    break;
082                case 5:
083                default:
084                    out = pin5;
085            }
086            // This concise one-liner isolates the bit specified by the bitmask 
087            // and checks if it is equal to one. This boolean value turns the
088            // DIO pin on or off.
089            out.set((code & mask) == mask);
090            // Shift the mask left one bit so it will get the next bit the next
091            // time the loop runs.
092            mask <<= 1;
093        }
094    }
095}
